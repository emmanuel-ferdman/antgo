# 管线构建和发布

## 创建功能管线
```
from antgo.pipeline import *
import numpy as np
image = np.zeros((128,128,3), dtype=np.uint8)
placeholder['image'](image). \
    resize_op['image', 'resized_image'](out_size=(384,256)). \
    build(
        platform='linux/x86-64',
        project_config={
            'input': [('image', 'EAGLEEYE_SIGNAL_RGB_IMAGE')],
            'output': [
                ('resized_image', 'EAGLEEYE_SIGNAL_TENSOR')
            ],
            'name': 'my',
            'git': '',
            'mode': 'server',           # 服务模式（server/none）
            'config': {                 # 管线配置，涉及管线参数（model_folder, 等），数据源设置
                'server_params': [{"node": "node_name", "name": "param_name", "value": "param_value", "type": "string"/"float"/"double"/"int"/"bool"}],
                'data_source': [{"type": "camera", "address": "", "format": "RGB/BGR", "mode": "NETWORK/USB/ANDROID_NATIVE/V4L2", "flag": "front"}]
            },
            'tool': {                   # protobuf工具位置
                'proto': '/workspace/grpc/cmake/build/.local'
            }
        },
        eagleeye_config={               # 依赖第三方库
            'ffmpeg': '/root/.3rd/ffmpeg/ffmpeg',
            'rk': '/root/.3rd/rk'
        }
    )
```

## 打包功能管线
```
from antgo.pipeline import *
# 基于管线名字作为参数，将所有关联项打包在一起
package("my")
```

目录结构如下，
```
package
    - config
        - plugin_config.json
    - dependents
        - ...so
    - model
    - plugins
        - xx
            libxx.so
    xx_demo
```

## 管线服务打包
```
# name 自定义服务名字
# main 入口文件名
# port 服务端口
# version antgo 版本分支
# image-repo 镜像中心服务（推荐使用阿里云镜像中心）
# user 镜像中心用户名（推荐使用阿里云镜像中心）
# password 镜像中心密码（推荐使用阿里云镜像中心）
# mode 打包模式（http, grpc, app）
antgo package --name=personseg --main=server --port=8080 --version=semib --image-repo=xxx --user=xxx --password=xxx
```

## 管线服务发布
```
# ip 目标机器IP地址
# port 对外服务端口
# user 镜像中心用户名（推荐使用阿里云镜像中心）
# password 镜像中心密码（推荐使用阿里云镜像中心）
antgo deploy --ip=xxx --port=xxx --user=xxx --password=xxx
```
